const getStakeholdersByGlobalPractice = () => { const combined = [ ...systemsData.map(i => ({ stakeholder: i.stakeholder_label, practice: i.global_practice })), ...datasetsData.map(i => ({ stakeholder: i.stakeholder_label, practice: i.global_practice })) ]; const unique = []; const seen = new Set(); combined.forEach(item => { const key = `${item.stakeholder}-${item.practice}`; if (!seen.has(key) && item.stakeholder && item.practice) { seen.add(key); unique.push(item); } }); const groups = {}; unique.forEach(item => { if (!groups[item.practice]) groups[item.practice] = new Set(); groups[item.practice].add(item.stakeholder); }); return Object.entries(groups).map(([practice, stakeholders]) => ({ practice: practice.length > 25 ? practice.substring(0, 22) + '...' : practice, count: stakeholders.size })).sort((a, b) => b.count - a.count); // Sort in descending order }; const getUniqueStakeholderPracticeData = () => { const combined = [ ...systemsData.map(i => ({ stakeholder_label: i.stakeholder_label, global_practice: i.global_practice })), ...datasetsData.map(i => ({ stakeholder_label: i.stakeholder_label, global_practice: i.global_practice })) ]; const unique = []; const seen = new Set(); combined.forEach(item => { const key = `${item.stakeholder_label}-${item.global_practice}`; if (!seen.has(key) && item.stakeholder_label && item.global_practice) { seen.add(key); unique.push(item); } }); // Sort by global_practice first, then by stakeholder_label return unique.sort((a, b) => { if (a.global_practice !== b.global_practice) { return a.global_practice.localeCompare(b.global_practice); } return a.stakeholder_label.localeCompare(b.stakeholder_label); }); };
